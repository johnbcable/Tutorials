Responsive

.header { grid-area: header; }
.menu { grid-area: menu; }
.hero { grid-area: hero; }
.main { grid-area: main; }
.banner { grid-area: banner; }
.extra { grid-area: extra; }
.image { grid-area: image; }

.grid {
  @media (min-width: 721px) {
    display: grid;
    grid-template-columns: repeat(4, 1fr);
    grid-template-rows: 40px 2fr repeat(4, 1fr);
    grid-template-areas:
    "header header header header"
    "hero hero hero hero"
    "menu main main main"
    "menu main main main"
    "menu banner banner banner"
    "menu extra image image";
  }

  @media (min-width: 1001px) {
    grid-template-areas:
      "header menu menu menu"
      "hero hero hero hero"
      "hero hero hero hero"
      "main main . image"
      "main main . extra"
      "banner banner banner banner";
  }
}

#fallback {
  //DEFINE MOBILE SPECIFIC STYLES HERE

  .grid {
    @media (min-width: 721px) {
      display: grid;
      grid-template-columns: repeat(4, 1fr);
      grid-template-rows: 40px 2fr repeat(4, 1fr);
      grid-template-areas:
        "header header header header"
        "hero hero hero hero"
        "menu main main main"
        "menu main main main"
        "menu banner banner banner"
        "menu extra image image";
    }

    @media (min-width: 1001px) {
      grid-template-areas:
        "header menu menu menu"
        "hero hero hero hero"
        "hero hero hero hero"
        "main main . image"
        "main main . extra"
        "banner banner banner banner";
    }
  }
  
  // Browsers that dont support grid
  @supports not (display:grid) {
    @media (min-width: 721px) {
      .header,
      .extra {
        float: left;
        width: 50%;
      }
    }
  }
  
  // Browsers that dont support grid nor '@supports' (IE10, 11)
  @media (min-width: 721px) and (-ms-high-contrast: none), (-ms-high-contrast: active) {
    .header,
    .extra {
      float: left;
      width: 50%;
    }
  }
}
